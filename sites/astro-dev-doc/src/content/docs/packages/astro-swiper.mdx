	---
  <Swiper
    options={{    // check options at https://swiperjs.com/swiper-api
      autoplay: {
        delay: 700,
        disableOnInteraction: false,
        waitForTransition: false,
      },
      loop: true,
    }}>
    <SwiperWrapper>

      <SwiperSlide> <img src=... /> </SwiperSlide>

    </SwiperWrapper>
